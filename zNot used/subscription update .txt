

        elif event['type'] == 'customer.subscription.updated':
            from datetime import datetime
            import json

            subscription = event['data']['object']
            stripe_customer_id = subscription.get('customer')
            cancel_at_end = subscription.get("cancel_at_period_end")

            items_data = subscription.get('items', {}).get('data', [])
            ends_at = items_data[0].get("current_period_end") if items_data else None
            subscription_end_date = (
                datetime.utcfromtimestamp(ends_at).isoformat()
                if ends_at else datetime.utcnow().isoformat()
            )

            if not stripe_customer_id:
                print(f"âš ï¸ Missing customer in subscription.updated event.")
                return '', 200

            if ends_at is None:
                print(f"âš ï¸ Ends_at was missing, using current UTC time instead â†’ {subscription_end_date}")

            price_to_plan = {
                "price_1RSmUX2MKajKZrXPjg7pnMCf": "pro",
                "price_1RSlus2MKajKZrXPHsYpvBYE": "plus",
            }

            price_id = items_data[0]["price"]["id"] if items_data else None
            new_plan = price_to_plan.get(price_id, "unknown")

            if subscription.get("schedule"):
                try:
                    schedule_id = subscription["schedule"]
                    print(f"ğŸ“‹ Fetching schedule: {schedule_id}")
                    schedule_obj = stripe.SubscriptionSchedule.retrieve(schedule_id)
                    future_phases = schedule_obj.get("phases", [])
                    print(f"ğŸ“‹ Full schedule phases object:\n{json.dumps(future_phases, indent=2)}")

                    now_unix = int(datetime.utcnow().timestamp())
                    scheduled_plan = None

                    for phase in future_phases:
                        if phase.get("start_date", 0) > now_unix:
                            future_price_id = phase["items"][0]["price"]
                            scheduled_plan = price_to_plan.get(future_price_id, "unknown")
                            print(f"ğŸ§  Detected future scheduled plan: {scheduled_plan}")
                            break

                    if scheduled_plan and scheduled_plan != new_plan:
                        new_plan = scheduled_plan
                        print(f"ğŸ“… Scheduled switch to: {new_plan} at next billing cycle.")
                    else:
                        print(f"ğŸŸ¡ No future scheduled plan change. Keeping plan as: {new_plan}")

                except Exception as e:
                    print(f"âš ï¸ Failed to retrieve scheduled plan: {e}")

            print(f"ğŸ” Final resolved plan for webhook: {new_plan}")
            print(json.dumps(subscription, indent=2))

            conn = sqlite3.connect('newsletter.db')
            cursor = conn.cursor()
            cursor.execute("SELECT id, plan FROM users WHERE stripe_customer_id = ?", (stripe_customer_id,))
            row = cursor.fetchone()

            if not row:
                print(f"âš ï¸ No user found for Stripe customer {stripe_customer_id}")
                conn.close()
                return '', 200

            user_id, current_plan = row
            if user_id is None:
                print("âš ï¸ Fetched user_id is None")
                conn.close()
                return '', 200

            user_id = int(user_id)
            print(f"ğŸ” Plan change detected: {current_plan} â†’ {new_plan}, cancel_at_end={cancel_at_end}")

            cursor.execute("""
                UPDATE users 
                SET subscription_end_date = ?
                WHERE id = ?
            """, (subscription_end_date, user_id))

            if cancel_at_end is False:
                print("ğŸ§ª Branch: cancel_at_end is False")
                if new_plan == current_plan:
                    print("ğŸ§ª Sub-branch: No plan change")
                    cursor.execute("""
                        UPDATE users
                        SET downgrade_to = NULL
                        WHERE id = ?
                    """, (user_id,))
                    print(f"ğŸš« downgrade_to cleared for user {user_id}")
                else:
                    print("ğŸ§ª Sub-branch: Plan changed")
                    cursor.execute("""
                        UPDATE users 
                        SET downgrade_to = ?
                        WHERE id = ?
                    """, (new_plan, user_id))
                    print(f"ğŸ“¦ downgrade_to set to {new_plan} for user {user_id}")

            elif cancel_at_end:
                print("ğŸ§ª Branch: cancel_at_end is True")
                cursor.execute("""
                    UPDATE users 
                    SET downgrade_to = ?
                    WHERE id = ?
                """, ("free", user_id))
                print(f"ğŸ“… Cancel scheduled: user {user_id} â†’ free on {subscription_end_date}")

            elif new_plan != current_plan and new_plan != "unknown":
                print("ğŸ§ª Branch: new_plan != current_plan and new_plan != 'unknown'")
                cursor.execute("""
                    UPDATE users 
                    SET downgrade_to = ?
                    WHERE id = ?
                """, (new_plan, user_id))
                print(f"ğŸ“¦ Plan switched without cancel: user {user_id} â†’ {new_plan} stored in downgrade_to")

            else:
                print(f"â„¹ï¸ Branch: No meaningful change detected")

            # âœ… FINAL CHECK â€” Apply scheduled downgrade immediately if it exists
            cursor.execute("SELECT downgrade_to FROM users WHERE id = ?", (user_id,))
            downgrade_row = cursor.fetchone()
            downgrade_to = downgrade_row[0] if downgrade_row else None

            if downgrade_to:
                print(f"ğŸ•’ Final check: applying scheduled downgrade_to={downgrade_to} for user {user_id}")
                cursor.execute("""
                    UPDATE users 
                    SET plan = ?, downgrade_to = NULL
                    WHERE id = ?
                """, (downgrade_to, user_id))
                print(f"âœ… Plan downgraded immediately to {downgrade_to} and downgrade_to cleared.")

            conn.commit()
            conn.close()
